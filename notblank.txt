// --- TEST SIMULATION: User moving in a circle ---
    const radius = 0.001;
const steps = 60;
let step = 0;

function simulateCircularMovement() {
    const angle = (step / steps) * 2 * Math.PI;
    const lat = startLat + radius * Math.cos(angle);
    const lon = startLon + radius * Math.sin(angle);

    // Move the user marker
    userMarker.setLatLng([lat, lon]);

    // Update trail
    updateTrail(lat, lon);

    // Emit location
    socket.emit("update_location", {
        map_id: mapID,
        latitude: lat,
        longitude: lon
    });

    step = (step + 1) % steps; // increment step

    // Only close loop when a full circle is complete
    if (step === 0 && trailPoints.length > 0) {
        socket.emit("territory_created", { map_id: mapID, coordinates: trailPoints.slice() });
        trailPoints = [];
        if (trailLine) {
            map.removeLayer(trailLine);
            trailLine = null;
        }
    }
}

// Run every 500ms
setInterval(simulateCircularMovement, 500);
